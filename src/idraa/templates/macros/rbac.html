{# RBAC visibility macros — gate template fragments behind a role allowlist without scattering ``current_user.role.value == "x"`` checks across every template that needs them. Usage: {% from "macros/rbac.html" import analyst_or_admin, admin_only %} {% call analyst_or_admin(current_user) %} Edit {% endcall %} {% call admin_only(current_user) %} Deactivate {% endcall %} The macros emit nothing when ``user`` is None or has the wrong role — matches the visual semantics of the legacy ``{% if current_user and current_user.role.value == "x" %}`` blocks they replace. They are visibility-only — server-side route checks (``Depends(require_role(...))``) remain the authority on writes. #} {%- macro analyst_or_admin(user) -%} {%- if user and user.role.value in ("analyst", "admin") -%} {{ caller() }} {%- endif -%} {%- endmacro -%} {%- macro admin_only(user) -%} {%- if user and user.role.value == "admin" -%} {{ caller() }} {%- endif -%} {%- endmacro -%}
